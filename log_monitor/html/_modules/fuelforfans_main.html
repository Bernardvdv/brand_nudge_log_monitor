[docs]def get_data(urls, country_code):
    """
    Makes a request to each URL and store the collected info in the database

    :param urls: Product url
    :param country_code: Country Code
    :return: True if successful and product count
    :rtype: Boolean and integer
    """
    try:
        insert_query = """
        INSERT INTO public.fuelforfans(
            date, retailer, "countryCode", currency, "sourceId", ean, brand, title, "shelfPrice", "wasPrice",
            "cardPrice", "inStock", "onPromo", "promoData", "skuURL", "imageURL", bundled, "masterSku",
            description, category, "priceValidUntil")
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s);
            """

        for item in urls:
            counter = 0
            full_url = BASE_URL + item
            r = Utils.make_request(full_url, HEADERS, PROXIES, RETAILER, False)
            if r and r.status_code == 200:
                soup = BeautifulSoup(r.text, "lxml")
                schema_products = soup.find_all("script", type='application/ld+json')
                # Because I keep forgetting, this is to find the product schema
                product_schema = Utils.get_product_schema(schema_products, "sku")
                if product_schema:
                    try:
                        product_list = json.loads(product_schema)
                    except:
                        product_list = None

                    if product_list:
                        source_id = product_list.get("sku", None)
                        title = product_list.get("name", None)
                        ean = product_list.get("mpn", None)
                        image_url = product_list.get("image", None)[0]
                        description = product_list.get("description", None)
                        category_div = soup.find("ol", {"class": "breadcrumb"})
                        category_urls = category_div.find_all("a")
                        category_list = []

                        for href in category_urls:
                            href_text = href.text.replace("\n", "")
                            if "Home" not in href_text:
                                category_list.append(href_text)
                        category = "/".join(category_list)

                        in_stock_schema = product_list["offers"].get("availability", None)
                        if "InStock" in in_stock_schema:
                            in_stock = True
                        else:
                            in_stock = False

                        currency = product_list["offers"].get("priceCurrency", None)
                        sku_url = product_list["offers"].get("url", None)
                        price_valid_until = product_list["offers"].get("priceValidUntil", None)
                        shelf_price = product_list["offers"].get("price", None)

                        try:
                            was_price = soup.find("del", {"class": "strike-through"})["content"]
                        except:
                            was_price = None

                        try:
                            promo_data = [soup.find("div", {"class": "callout"}).text.replace("\n", "")]
                            if promo_data:
                                on_promo = True
                            else:
                                on_promo = False
                        except:
                            promo_data = []
                            on_promo = False
                        try:
                            brand = soup.find("div", {"class": "product-brand"}).text
                        except:
                            brand = None

                        date = datetime.now()
                        card_price = None
                        bundled = None
                        master_sku = None

                        counter += 1

                        row_data = [date, RETAILER, country_code, currency, source_id, ean, brand,
                                    title, shelf_price, was_price, card_price, in_stock, on_promo,
                                    json.dumps(promo_data), sku_url, image_url, bundled, master_sku,
                                    description, category, price_valid_until]

                        Utils.insert_to_db(insert_query, row_data, RETAILER)

        return True, counter
    except Exception as e:
        # Log to DB
        log_error.error_db(RETAILER, e, sys.exc_info())
        # Log to file
        log_error.error_file(RETAILER, e, sys.exc_info())
        return False




[docs]def get_product_urls(url):
    """
    Finds all the product hyperlinks and remove duplicates

    :param url: Category URL
    :return: All product urls
    :rtype: List
    """
    prox = Proxy()
    prox.socks_version = 5
    prox.proxy_type = ProxyType.MANUAL
    prox.http_proxy = os.getenv("PROXIES_HTTPS")
    prox.socks_proxy = os.getenv("PROXIES_HTTPS")
    prox.ssl_proxy = os.getenv("PROXIES_HTTPS")

    capabilities = webdriver.DesiredCapabilities.CHROME
    prox.add_to_capabilities(capabilities)

    options = Options()
    options.add_argument("--no-sandbox")
    options.add_argument("window-size=1920,1080")
    options.headless = True
    driver = webdriver.Chrome(executable_path="chromedriver", options=options, desired_capabilities=capabilities)

    driver.get(url)
    time.sleep(2)  # Allow 2 seconds for the web page to open
    scroll_pause_time = 1
    screen_height = driver.execute_script("return window.screen.height;")
    i = 1
    driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

    time.sleep(3)
    for x in range(0, 20):
        try:
            all_results_button = WebDriverWait(driver, 30).until(
                EC.element_to_be_clickable((By.CLASS_NAME, "js-show-all")))
            driver.execute_script('arguments[0].click()', all_results_button)
            break
        except:
            time.sleep(3)
            continue

    while True:
        # scroll one screen height each time
        driver.execute_script("window.scrollTo(0, {screen_height}*{i});".format(screen_height=screen_height, i=i))
        i += 1
        time.sleep(scroll_pause_time)
        # update scroll height each time after scrolled, as the scroll height can change after we scrolled the page
        scroll_height = driver.execute_script("return document.body.scrollHeight;")
        # Break the loop when the height we need to scroll to is larger than the total scroll height
        if screen_height * i > scroll_height:
            break

    soup = BeautifulSoup(driver.page_source, "html.parser")
    product_grid = soup.find("div", {"class": "product-grid"})
    if product_grid:
        items_grid = product_grid.find_all("a", {"class": "product-tile-image"})
        if items_grid:
            grid_urls = []
            for item in items_grid:
                url = item['href']
                grid_urls.append(url)

    return grid_urls




[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            # Cleaned_urls returns all the products urls on the page
            cleaned_urls = get_product_urls(url["url"])
            result = get_data(cleaned_urls, url["country"])
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter,
                          navigation_url_count,
                          item_counter,
                          duration,
                          urls,
                          RETAILER,
                          TABLE_NAME)
