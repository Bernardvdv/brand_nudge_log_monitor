    def error_db(retailer, message, sys_info):
        """
        Handler to insert error messages to the database

        :param retailer: Retailer name
        :param message: Error message from exception
        :param sys_info: Error object
        :return: None
        :rtype: None
        """
        try:
            tb = sys_info[-1]
            stk = traceback.extract_tb(tb, 1)
            function_name = stk[0][2]
            lineno = tb.tb_lineno
            error_record = {
                "Created": datetime.now(),
                "Name": retailer,
                "Level": 1,
                "Message": str(message),
                "Function": function_name,
                "LineNo": lineno
            }

            connection = Utils.create_db_con()
            connection_cursor = connection.cursor()
            # Let's check if the log table exist and create if not
            LogHandler.create_log_table(connection)
            connection_cursor.execute(constants.insertion_sql, error_record)

            connection.commit()
            connection.cursor().close()
        except:
            pass

    def cron_logs(lines):
        """
        Log cron.log entries to the DB table

        :param lines: cron file content
        :return: None
        :rtype: None
        """
        try:
            connection = Utils.create_db_con()
            connection_cursor = connection.cursor()
            # Let's check if the cron table exist and create if not
            LogHandler.create_cron_table(connection)
            for line in lines:
                data = [datetime.now(), line]
                connection_cursor.execute(constants.cron_insertion, data)
                connection.commit()
            connection.cursor().close()
        except Exception as e:
            # Log to DB
            LogHandler.error_db("cron", e, sys.exc_info())
            # Log to file
            LogHandler.error_file("cron", e, sys.exc_info())
            pass

    def statistics(navigation_counter, navigation_url_count, item_counter, duration, urls, retailer, table_name):
        """
        Information on expected and collected data

        :param navigation_counter: Amount of navigation URLs accessed
        :param navigation_url_count: Expected navigation URLs
        :param item_counter: Amount of items collected
        :param duration: Time the scraper took to complete
        :param urls: URL dictionary post scraping used for statistics
        :param retailer: Retailer collected
        :param table_name: SQL table name where data are inserted
        :return: None
        :rtype: None
        """
        failed_urls = list()
        for url in urls:
            if url["success"] is False:
                failed_urls.append(url["url"])
        stats = {
            "Navigation_count": navigation_counter,
            "Navigation_url_count": navigation_url_count,
            "Item_counter": item_counter,
            "Duration": duration,
            "Failed_URLS": failed_urls,
            "Table_Name": table_name
        }
        # Log to Slack
        LogHandler.error_slack(retailer, stats, "stats")
        # Log to DB
        Statistics.log_to_db(retailer, stats)

    def time_format(duration):
        """
        Convert duration provided to friendly readable time
        e.g. 01H 08m 55s

        :param duration: Duration in millisecond
        :return: Reformatted time
        :rtype: String
        """
        if duration is not None:
            seconds = int(duration)
            d = seconds // (3600 * 24)
            h = seconds // 3600 % 24
            m = seconds % 3600 // 60
            s = seconds % 3600 % 60
            if d > 0:
                return '{:02d}D {:02d}H {:02d}m {:02d}s'.format(d, h, m, s)
            elif h > 0:
                return '{:02d}H {:02d}m {:02d}s'.format(h, m, s)
            elif m > 0:
                return '{:02d}m {:02d}s'.format(m, s)
            elif s > 0:
                return '{:02d}s'.format(s)
        return "N/a"

    def make_request(url, request_headers, proxy, retailer, verify=True):
        """
        Request made to provided URL by using provided headers, proxy and SSL verification

        :param url: Target URL
        :param request_headers: HTTP header used in the HTTP request to provide information
        :param proxy: Enabled True else false
        :param retailer: Retailer collected
        :param verify: whether we verify the server's TLS certificate
        :return: Response object
        :rtype: Object
        """
        try:
            for n in range(constants.MAX_RETRIES):
                r = requests.request(
                    "GET",
                    url,
                    proxies=proxy,
                    headers=request_headers,
                    timeout=20,
                    allow_redirects=True,
                    verify=verify
                )
                if r and r.status_code == 200:
                    return r
                else:
                    continue
        except Exception as e:
            # Log to DB
            LogHandler.error_db(retailer, e, sys.exc_info())
            # Log to file
            LogHandler.error_file(retailer, e, sys.exc_info())

    def get_product_schema(schema_products, search_term):
        """
        Loops through available schema's to find the product object

        :param schema_products: Available json schema's
        :param search_term: Value to look for in schema to find the product dict
        :return: product schema
        :rtype: String
        """
        for schema in schema_products:
            schema = str(schema)
            cleaned_schema_products = schema.replace(
                '<script type="application/ld+json">', ""
            ).replace("</script>", "")

            if search_term in cleaned_schema_products:
                return cleaned_schema_products
        return None
