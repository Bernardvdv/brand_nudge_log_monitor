[docs]def cleanup():
    """
    Removing all duplicates for the current date which matches the where clause criteria
    :return: None
    """

    connection = Utils.create_db_con()
    cursor = connection.cursor()

    duplicate_delete_query = """
    DELETE FROM
        public.dunnes_store a
        USING public.dunnes_store b
        WHERE
            a.id < b.id
            AND a."skuURL" = b."skuURL"
            AND a."sourceId" = b."sourceId"
            AND a."shelfPrice" = b."shelfPrice"
            AND a."title" = b."title"
            AND a.date = now()::timestamp::date;
    """

    try:
        cursor.execute(duplicate_delete_query)
        connection.commit()
    except Exception as e:
        # Log to DB
        log_error.error_db(retailer, e, sys.exc_info())
        # Log to file
        log_error.error_file(retailer, e, sys.exc_info())
        pass




[docs]def get_navigation_urls():
    """ Get all navigation URLs from the state API
    :returns URLs as a dictionary
    """
    r = Utils.make_request(DUNNES_STORE_BASE_URL, headers, False, retailer)
    soup = BeautifulSoup(r.text, "lxml")
    menu_content = soup.select('a[class*="CategoryHiddenLink-"]')
    item_list = []
    for item in menu_content:
        url_list_item = {
                "url": item["href"],
                "text": item.text,
                "attempts": 0,
                "max_attempts": MAX_RETRIES,
                "success": False
        }
        item_list.append(url_list_item)
    return item_list

    try:
        r = Utils.make_request(url, headers, False, retailer)
        # print(f"Current URL: {url} :::::: Status Code: {r.status_code}")
        soup = BeautifulSoup(r.text, "lxml")
        product_grid = soup.find("a", {"data-testid": "link-testId"}).text
        item_counter = 0
        # We skip this URL as it is a carousel and items will be scraped on their individual URLS
        if "View all" not in product_grid:
            listing_regex = re.compile('.*ColListing--.*')
            for item in soup.find_all("div", {"class": listing_regex}):
                item_counter += 1
                skuURL = item.find("a", {"class": re.compile(".*ProductCardHiddenLink--.*")})["href"]
                promotion_badges = item.find("div", {"class": re.compile(".*PromotionBadges--.*")}).find("div")

                if promotion_badges:
                    badges = promotion_badges.find_all("div")
                    promoData = list()
                    for badge in badges:
                        promoData.append(badge.text)
                    if len(promoData) > 0:
                        onPromo = True
                    else:
                        onPromo = False
                else:
                    promoData = None
                    onPromo = False

                shelfPrice = item.find("span", {"class": re.compile(".*ProductCardPrice--.*")}).text[1:]
                try:
                    try:
                        wasPrice = item.find("span", {"class": re.compile(".*WasPrice--.*")})
                        if wasPrice:
                            wasPrice = wasPrice.text
                    except:
                        wasPrice = None
                    if wasPrice:
                        wasPrice = wasPrice.replace(" was", "").strip()[1:]
                    else:
                        # try and get it from the promo
                        if promoData:
                            for promo in promoData:
                                if "SAVE" in promo:
                                    savings_value = promo.replace("SAVE €", "")
                                    wasPrice = int(shelfPrice) + int(savings_value)
                except:
                    wasPrice = None

                product_id = skuURL[skuURL.rindex('-') + 1:]
                prod_api_url = f"https://storefrontgateway.dunnesstoresgrocery.com/api/stores/524/products/{product_id}"
                product_info = Utils.make_request(prod_api_url, headers, False, retailer)
                if product_info and product_info.status_code == 200:
                    product_data = product_info.json()
                    try:
                        title = product_data.get("name", None)
                    except:
                        title = None
                    try:
                        category = product_data.get("categories", None)
                        cat_list = list()
                        if category:
                            for cat in category:
                                cat_list.append(cat.get("category"))
                            category = "/".join(cat_list)
                        else:
                            category = None
                    except:
                        category = None
                    try:
                        sourceId = product_data.get("sku", None)
                    except:
                        sourceId = None
                    try:
                        description = Utils.remove_html_tags(product_data.get("description", None))
                    except:
                        description = None
                    try:
                        brand = product_data.get("brand", None)
                    except:
                        brand = None
                    try:
                        imageURL = product_data["primaryImage"].get("default")
                    except:
                        imageURL = None
                    try:
                        unitType = product_data["unitsOfSize"].get("label", None)
                    except:
                        unitType = None
                    try:
                        unitPrice = product_data.get("unitPrice", None)
                        if unitPrice:
                            unitPrice = unitPrice.replace("€", "")
                    except:
                        unitPrice = None
                    try:
                        inStock = product_data.get("available", None)
                        if inStock:
                            inStock = True
                        else:
                            inStock = False
                    except:
                        inStock = False

                    countryCode = "IE"
                    date = datetime.now()
                    cardPrice = None

                    masterSku = None

[docs]def parse_urls(urls):
    """
    Scrape each product URL and inserts to DB.
    :param urls: List obj of all product URLS
    :return: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0

    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        full_url = DUNNES_STORE_BASE_URL + url["url"]
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(full_url)
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter, duration, urls, retailer, TABLE_NAME)
