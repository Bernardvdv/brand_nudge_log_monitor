[docs]def get_description(schema):
    """
    Get the description from the provided javascript tag

    :param schema: Javascript tag containing the product data
    :return: Description if found else None
    :rtype: String
    """
    cleaned_schema_products = schema.replace(
        '<script id="__NEXT_DATA__" type="application/json">', ""
    ).replace("</script>", "")

    product_list = json.loads(cleaned_schema_products)
    if "product" in product_list[0]["props"]["pageProps"]:
        description = Utils.remove_html_tags(product_list[0]["props"]["pageProps"]["product"].get("description", None))
    elif "skus" in product_list[0]["props"]["pageProps"]:
        description = Utils.remove_html_tags(product_list[0]["props"]["pageProps"].get("productInformation"))
    else:
        description = None
    return description

[docs]def get_review_detail(schema):
    """
    Get the product review detail from the provided javascript tag

    :param schema: Javascript tag containing the product data
    :return: Review count, product rating if found else None
    :rtype: List
    """
    cleaned_schema_products = schema.replace(
        '<script id="__NEXT_DATA__" type="application/json">', ""
    ).replace("</script>", "")

    product_list = json.loads(cleaned_schema_products)

    try:
        if "product" in product_list[0]["props"]["pageProps"] and "reviewSummary" in \
                product_list[0]["props"]["pageProps"]["product"]:
            try:
                review_count = product_list[0]["props"]["pageProps"]["product"]["reviewSummary"].get("numberOfReviews")
            except:
                review_count = None
            try:
                product_rating = product_list[0]["props"]["pageProps"]["product"]["reviewSummary"].get("averageRating")
            except:
                product_rating: None
        elif "skus" in product_list[0]["props"]["pageProps"] and "reviewData" in product_list[0]["props"]["pageProps"]:
            try:
                review_count = product_list[0]["props"]["pageProps"]["reviewData"].get("numberOfReviews")
            except:
                review_count = None
            try:
                product_rating = product_list[0]["props"]["pageProps"]["reviewData"].get("averageRating")
            except:
                product_rating = None
        else:
            review_count = None
            product_rating = None
    except Exception as e:
        log_error.error_file(RETAILER, e, sys.exc_info())

    return review_count, product_rating

[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0

    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data_from_search_query(url["url"])
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter, duration, urls, RETAILER, TABLE_NAME)
