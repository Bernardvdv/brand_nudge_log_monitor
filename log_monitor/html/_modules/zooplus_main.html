[docs]def get_data(url):
    """
    Makes a request to each URL and store the collected info in the database

    :param url: Product url
    :param category: Product category
    :return: True if successful and product count
    :rtype: Boolean and integer
    """
    try:
        insert_query = """
            INSERT INTO public.zooplus(
                date, retailer, "countryCode", currency, "sourceId", ean, brand, title, "shelfPrice", "wasPrice",
                "cardPrice", "inStock", "onPromo", "promoData", "skuURL", "imageURL", bundled, "masterSku",
                description, "reviewCount", category)
                VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s);
            """

        item_counter = 0
        r = Utils.make_request(BASE_URL + url, HEADERS, False, RETAILER, False)
        if r and r.status_code == 200:
            soup = BeautifulSoup(r.text, "lxml")

            product_grid = soup.find("div", {"data-zta": "product-list-wrapper"})
            if product_grid:
                sub_menu_links = product_grid.find_all('div', {"data-zta": "product-box-list"})
                for item in sub_menu_links:
                    item_url = item.find("a", {"data-zta": "product-link"})["href"]
                    r = Utils.make_request(BASE_URL + item_url, HEADERS, False, RETAILER, False)
                    if r and r.status_code == 200:
                        soup = BeautifulSoup(r.text, "lxml")

                        product_variant_schema = soup.find_all("script", type='application/json')
                        for x in product_variant_schema:
                            schema = str(x)
                            if "__NEXT_DATA__" in str(schema):
                                cleaned_variant_products = schema.replace(
                                    '<script id="__NEXT_DATA__" type="application/json">', ""
                                ).replace("</script>", "")
                        if cleaned_variant_products:
                            variant_list = json.loads(cleaned_variant_products)
                            if "pageLevelProps" in variant_list["props"]["pageProps"]:
                                variants = variant_list["props"]["pageProps"]["pageLevelProps"]["productDetails"]
                                brand = variants["product"]["brand"]
                                image_url = variants["product"]["mediaDetails"]["medias"][0].get("full", None)
                                currency = variants["product"]["priceAggregate"].get("currency", None)
                                master_sku = variants["product"].get("productId")
                                description = Utils.remove_html_tags(variants["product"].get("description"))
                                review_count = variants["feedbackAggregate"].get("reviewCount")
                                for variant in variants["product"]["articleVariants"]:
                                    source_id = variant.get("id")
                                    is_set = variant["price"].get("type")
                                    if is_set:
                                        ean = f"{is_set}_{variant.get('ean13')}"
                                    else:
                                        ean = variant.get('ean13')
                                    title = variant.get("omTitle")
                                    shelf_price = variant["price"].get("currentPriceRaw")
                                    was_price = variant["price"].get("beforePrice")
                                    if was_price:
                                        was_price = was_price[1:]
                                    in_stock = variant["available"]
                                    on_promo = variant["whileStockLasts"]

                                    promo_data = []
                                    try:
                                        promo_p = soup.find("p", {"data-zta": "productDescription"})
                                        if promo_p:
                                            promo_font = promo_p.find("font")
                                            if promo_font:
                                                promo_data.append(promo_font.text)
                                                on_promo = True
                                            # promo_span = promo_p.find("span")
                                            # if promo_span:
                                            #     promo_data.append(promo_span.text)
                                            #     on_promo = True
                                    except:
                                        promo_data = []

                                    category_div = soup.find("ul", {"data-zta": "breadcrumbs"})
                                    category_urls = category_div.find_all("li")

                                    category_list = []
                                    for href in category_urls:
                                        if href.text != 'back to category':
                                            if href.text != 'zooplus.co.uk/':
                                                category_list.append(href.text.replace(" /", "").replace("/", "")
                                                                     .replace("â–º", ""))
                                    category_list.pop()
                                    category = "/".join(category_list)

                                    sku_url = BASE_URL + item_url
                                    country_code = "UK"
                                    date = datetime.now()
                                    card_price = None
                                    bundled = None

                                    item_counter += 1

                                    row_data = [date, RETAILER, country_code, currency, source_id, ean, brand, title,
                                                shelf_price, was_price, card_price, in_stock, on_promo,
                                                json.dumps(promo_data), sku_url, image_url, bundled, master_sku,
                                                description, review_count, category]

                                    Utils.insert_to_db(insert_query, row_data, RETAILER)
                        else:
                            item_counter = 0
        return True, item_counter
    except Exception as e:
        # Log to DB
        log_error.error_db(RETAILER, e, sys.exc_info())
        # Log to file
        log_error.error_file(RETAILER, e, sys.exc_info())
        return False




[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(url["url"])
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter,
                          navigation_url_count,
                          item_counter,
                          duration,
                          urls,
                          RETAILER,
                          TABLE_NAME)
