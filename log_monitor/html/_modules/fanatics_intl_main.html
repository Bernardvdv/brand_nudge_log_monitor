[docs]def get_data(item):
    """
    Makes a request to each URL and store the collected info in the database

    :param item: Product url
    :return: True if successful and product count
    :rtype: Boolean and intesger
    """
    try:
        insert_query = """
           INSERT INTO public.fanatics_intl(
            date, retailer, "countryCode", currency, "sourceId", ean, brand, title, "shelfPrice", "wasPrice",
            "cardPrice", "inStock", "onPromo", "promoData", "skuURL", "imageURL", bundled, "masterSku",
            description, category, "itemCondition", "productId", "availableSizes")
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s);
            """

        counter = 0
        r = Utils.make_request(item["url"], CUSTOM_HEADERS, PROXIES, RETAILER, False)

        if r and r.status_code == 200:
            soup = BeautifulSoup(r.text, "lxml")

            pagination_div = soup.find("div", {"data-talos": "pageCount"})
            if pagination_div:
                pagination_string = pagination_div.text
                match = re.search(r'\d+', pagination_string[::-1])  # search for digits from the end of the string
            if match:
                last_num = match.group()[::-1]
            else:
                last_num = 1

            url_list = []
            for url in range(1, int(last_num) + 1):
                full_url = f"{item['url']}&pageNumber={url}&sortOption=TopSellers"
                r = Utils.make_request(full_url, CUSTOM_HEADERS, PROXIES, RETAILER, False)
                if r and r.status_code == 200:
                    soup = BeautifulSoup(r.text, "lxml")

                    product_grid = soup.find("div", {"class": "product-grid-container"})
                    if product_grid:
                        products = product_grid.find_all("div", {"class": "product-image-container"})
                        for product in products:
                            href = product.find("a")["href"]
                            url_list.append(BASE_URL + href)

            for url in url_list:
                r = Utils.make_request(f"{url}?_ref=p-TLP%3Am-GRID%3Ai-r0c0%3Apo-0%3Fcur%3D{item['currency']}"
                                       f"&cur={item['currency']}", CUSTOM_HEADERS, PROXIES, RETAILER, False)
                if r and r.status_code == 200:
                    soup = BeautifulSoup(r.text, "lxml")
                    price_div = soup.find("div", {"class": "price-card"})
                    shelf_price_text = price_div.find("span", {"class": 'lowest'}).find(
                        "span", {"class": "sr-only"}).text

                    if "US" in shelf_price_text:
                        shelf_price = shelf_price_text.replace("US$", "")
                    elif "A$" in shelf_price_text:
                        shelf_price = shelf_price_text.replace("A$", "")
                    else:
                        shelf_price = shelf_price_text[1:]
                    was_price_span = price_div.find("span", {"class": "strike-through"})
                    try:
                        if was_price_span:
                            was_price_text = was_price_span.find("span", {"class": "sr-only"}).text
                            if "US" in was_price_text:
                                was_price = was_price_text.replace("US$", "")
                            elif "A$" in was_price_span.text:
                                was_price = was_price_text.replace("A$", "")
                            else:
                                was_price = was_price_text[1:]
                        else:
                            was_price = None
                    except:
                        was_price = None
                    if was_price:
                        on_promo = True
                    else:
                        on_promo = False
                    available_sizes_div = soup.find("div", {"class": "size-selector-list"})
                    if available_sizes_div:
                        available_sizes = []
                        for size in available_sizes_div:
                            available_sizes.append(size.text)
                        available_sizes = ",".join(available_sizes)
                    else:
                        available_sizes = None

                    category_div = soup.find("ul", {"class": "breadcrumbs-container"})
                    category_urls = category_div.find_all("a")
                    category_list = []

                    for href in category_urls:
                        category_list.append(href.text)
                    category = "/".join(category_list)

                    schema_products = soup.find_all("script", type='application/ld+json')
                    product_schema = Utils.get_product_schema(schema_products, "Product")
                    if product_schema:
                        try:
                            product_list = json.loads(product_schema)
                        except:
                            product_list = None

                        try:
                            title = product_list.get("name", None)
                        except:
                            title = None
                        try:
                            brand = product_list.get("brand", None)
                        except:
                            brand = None
                        try:
                            description = product_list.get("description", None)
                        except:
                            description = None
                        try:
                            image_url = product_list.get("image", None)[0]
                        except:
                            image_url = None
                        try:
                            item_condition = product_list.get("itemCondition", None)
                        except:
                            item_condition = None
                        try:
                            in_stock = product_list['offers'][0].get("availability", None)
                            if in_stock:
                                if in_stock == 'http://schema.org/InStock':
                                    in_stock = True
                                else:
                                    in_stock = False
                            else:
                                in_stock = False
                        except:
                            in_stock = False
                        try:
                            currency = product_list['offers'][0]['priceSpecification'].get("priceCurrency", None)
                        except:
                            None
                        try:
                            source_id = product_list['offers'][0].get("serialNumber", None)
                        except:
                            source_id = None
                        try:
                            sku_url = product_list.get("url", None)
                        except:
                            sku_url = None
                        try:
                            product_id = product_list.get("productID", None)
                        except:
                            product_id = None

                        country_code = item["country"]
                        date = datetime.now()
                        card_price = None
                        bundled = None
                        master_sku = None
                        ean = None
                        promo_data = []

                        counter += 1

                        row_data = [date, RETAILER, country_code, currency, source_id, ean, brand,
                                    title, shelf_price, was_price, card_price, in_stock, on_promo,
                                    json.dumps(promo_data), sku_url, image_url, bundled, master_sku,
                                    description, category, item_condition, product_id, str(available_sizes)]

                        Utils.insert_to_db(insert_query, row_data, RETAILER)

        return True, counter
    except Exception as e:
        # Log to DB
        log_error.error_db(RETAILER, e, sys.exc_info())
        # Log to file
        log_error.error_file(RETAILER, e, sys.exc_info())
        return False

[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(url)
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter,
                          navigation_url_count,
                          item_counter,
                          duration,
                          urls,
                          RETAILER,
                          TABLE_NAME)
