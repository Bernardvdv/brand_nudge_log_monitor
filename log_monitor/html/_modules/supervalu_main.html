[docs]def get_was_price(soup):
    """
    Get value of WAS from string
    :param soup: Product object
    :return: Was price value of none
    """
    try:
        promo_text = soup.find('div', {'class': 'product-details-promotion-name'}).find("span").text
        if "WAS" in promo_text.split():
            was, now = [value.strip() for value in promo_text.split(',')[:2]]
            was_value = was.split()[-1].replace("WAS ", "")
            return was_value
        else:
            return None
    except Exception as e:
        # Log to DB
        log_error.error_db(retailer, e, sys.exc_info())
        # Log to file
        log_error.error_file(retailer, e, sys.exc_info())
        pass

[docs]def get_navigation_urls():
    """ Get all navigation URLs from the state API
    :returns URLs as a dictionary
    """
    r = Utils.make_request(SUPERVALU_BASE_URL, headers, False, retailer)
    soup = BeautifulSoup(r.text, "lxml")
    # Get the schema
    urls = json.loads(soup.find('script', id='mm-items').text)
    item_list = []
    for item in urls:
        url_list_item = {
                "url": item["url"],
                "category": item["dn"],
                "attempts": 0,
                "max_attempts": MAX_RETRIES,
                "success": False
        }

        item_list.append(url_list_item)
    return item_list




[docs]def parse_urls(urls):
    """
    Create BS4 object from url
    :param urls: Product URLS to parse through
    :return:None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(url)
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter,
                          navigation_url_count,
                          item_counter,
                          duration,
                          urls,
                          retailer,
                          TABLE_NAME)
