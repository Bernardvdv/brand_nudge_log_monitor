    """
    Makes a request to each URL and store the collected info in the database

    :param url: Product url
    :return: True if successful and product count
    :rtype: Boolean and integer
    """
    try:
        pagesize = 200

        r = Utils.make_request(f"{url['url']}?pageSize={pagesize}&sort=relevance", HEADERS, False, RETAILER)
        if r and r.status_code == 200:
            soup = BeautifulSoup(r.text, "lxml")
            try:
                last_pagination_value_text = soup.find("div", {"class": "pagination-bar-results"}).text
                number_of_items = re.findall('\d+', last_pagination_value_text)[0]
                last_pagination_value = math.ceil(int(number_of_items) / pagesize)
            except:
                last_pagination_value = 1

            item_counter = 0
            for page in range(0, int(last_pagination_value) + 1):
                full_url = f"{url['url']}?sort=relevance&page={page}&pageSize={pagesize}"
                r = Utils.make_request(full_url, HEADERS, False, RETAILER)
                if r and r.status_code == 200:
                    soup = BeautifulSoup(r.text, "lxml")

                    product_grid = soup.find("div", {"class": "product__grid"})
                    products = product_grid.find_all("div", {"class": "product-item"})
                    for product in products:
                        sku_url = BASE_URL + product.find("a", {"class": "product-link"})["href"]
                        r = Utils.make_request(sku_url, HEADERS, False, RETAILER)
                        if r and r.status_code == 200:
                            soup = BeautifulSoup(r.text, "lxml")
                            product_info = soup.find("div", {"class": "product-main-info"})

                            shelf_price = product_info.get('data-price-formatted-value', None)[1:]
                            source_id = product_info.get('data-sku', None)
                            title = product_info.get('data-product-name', None)
                            in_stock = product_info.get('data-stock-level-status-code', None)
                            if in_stock and "inStock" in in_stock:
                                in_stock = True
                            else:
                                in_stock = False
                            try:
                                was_price = soup.find("span", {"class": "pdp-was-price__value"}).text.replace(
                                    "\n", "").replace("\t", "")[1:]
                            except:
                                was_price = None
                            try:
                                promo_data_label = soup.find("span", {"class": "pdp-you-save-price__label"}).text. \
                                    replace("\n", "").replace("\t", "")
                                promo_data_value = soup.find("span", {"class": "pdp-you-save-price__value"}).text. \
                                    replace("\n", "").replace("\t", "")
                                promo_data = f"{promo_data_label} {promo_data_value}"
                            except:
                                promo_data_label = None
                                promo_data = []
                            if promo_data_label in promo_data:
                                on_promo = True
                            else:
                                on_promo = False
                            category_div = soup.find("ol", {"class": "breadcrumb"})
                            category_urls = category_div.find_all("a")
                            category_list = []

                            for href in category_urls:
                                crumb = href.text.replace("\n", "")
                                if not crumb == "":
                                    category_list.append(crumb)
                            category = "/".join(category_list)
                            image_url = BASE_URL + soup.find("div", {"class": "pdp-image-gallery__item"}).find(
                                "img")["src"]

                            specification = soup.find_all("div", {"class": "specification-item"})

                            brand = None
                            ean = None

                            for spec in specification:
                                spec_text = spec.find("span", {"class": "specification-name"}).text.replace(
                                    "\n", "").strip()
                                if spec_text == "Brand:":
                                    brand = spec.find("span", {"class": "specification-value"}).text.replace(
                                        "\n", "").strip()
                                if spec_text == "Bar Code:":
                                    ean = spec.find("span", {"class": "specification-value"}).text.replace(
                                        "\n", "").strip()

                            country_code = "UK"
                            currency = "GBP"
                            date = datetime.now()
                            card_price = None
                            bundled = None
                            master_sku = None

                            insert_query = """
                                        INSERT INTO public.wilko(
                                            date, retailer, "countryCode", currency, "sourceId", ean, brand, title,
                                            "shelfPrice", "wasPrice", "cardPrice", "inStock", "onPromo", "promoData",
                                            "skuURL", "imageURL", bundled, "masterSku", category)
                                            VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s,
                                            %s, %s, %s);
                                        """

                            row_data = [date, RETAILER, country_code, currency, source_id, ean,
                                        brand, title, shelf_price, was_price, card_price, in_stock,
                                        on_promo, json.dumps(promo_data), sku_url, image_url, bundled,
                                        master_sku, category]
                            item_counter += 1

                            Utils.insert_to_db(insert_query, row_data, RETAILER)
            return True, item_counter
    except Exception as e:
        # Log to DB
        log_error.error_db(RETAILER, e, sys.exc_info())
        # Log to file
        log_error.error_file(RETAILER, e, sys.exc_info())
        return False




[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    navigation_url_count = 0

    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(url)
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter,
                          duration, urls, RETAILER, TABLE_NAME)
