[docs]def get_data(urls):
    """
    Makes a request to each URL and store the collected info in the database

    :param urls: Product url
    :return: True if successful and product count
    :rtype: Boolean and integer
    """
    try:
        item_counter = 0
        for url in urls:
            full_url = BASE_URL + url
            r = Utils.make_request(full_url, HEADERS, False, RETAILER)
            if r and r.status_code == 200:
                soup = BeautifulSoup(r.text, "lxml")
                if soup:
                    schema_products = soup.find_all("script", type="application/ld+json")
                    for x in schema_products:
                        schema = str(x)
                        if "sku" in str(schema):
                            cleaned_schema_products = schema.replace(
                                '<script type="application/ld+json">', ""
                            ).replace("</script>", "")

                    try:
                        product_list = json.loads(cleaned_schema_products)
                    except:
                        product_list = None

                    if product_list:
                        try:
                            title = product_list.get("name", None)
                        except:
                            title = None
                        try:
                            image_url = product_list.get("image", None)
                        except:
                            image_url = None
                        try:
                            description = product_list.get("description", None)
                        except:
                            description = None
                        try:
                            source_id = product_list.get("sku", None)
                        except:
                            source_id = None
                        try:
                            brand = product_list["brand"].get("name", None)
                        except:
                            brand = None
                        try:
                            currency = product_list["offers"].get("priceCurrency", None)
                        except:
                            currency = None
                        try:
                            shelf_price = product_list["offers"].get("price", None)
                        except:
                            shelf_price = None
                        try:
                            price_valid_until = product_list["offers"].get("priceValidUntil", None)
                        except:
                            price_valid_until = None
                        try:
                            item_condition = product_list["offers"].get("itemCondition", None)
                        except:
                            item_condition = None
                        try:
                            in_stock = product_list["offers"].get("availability", None)
                            if in_stock:
                                if "InStock" in in_stock:
                                    in_stock = True
                                elif "OutOfStock" in in_stock:
                                    in_stock = False
                                else:
                                    in_stock = False
                            else:
                                in_stock = False
                        except:
                            in_stock = False
                        try:
                            seller = product_list["offers"]["seller"].get("name", None)
                        except:
                            seller = None

                        # Get Promos
                        promodiv = soup.find("div", itemprop="offers")
                        if promodiv:
                            promo_data = list()
                            promodivtag = promodiv.find_all("span", {"class": "bm-small"})
                            if promodivtag:
                                was_price = promodivtag[0].text.replace("WAS ", "")[1:]
                                for promo in promodivtag:
                                    promo_data.append(promo.text)
                                on_promo = True
                            else:
                                was_price = None
                                promo_data = []
                                on_promo = False
                        else:
                            was_price = None
                            promo_data = []
                            on_promo = False

                        category_div = soup.find("div", {"align-items-stretch"}).find("ul", {"class": "list-inline"})
                        category_urls = category_div.find_all("a", {"class": "text-dark"})
                        category_list = []
                        for href in category_urls:
                            if "Home" not in href.text:
                                category_list.append(href.text)
                        category_list.pop()
                        category = "/".join(category_list)

                        country_code = "UK"
                        date = datetime.now()
                        card_price = None
                        bundled = None
                        master_sku = None
                        ean = None

                        item_counter += 1

                        insert_query = """
                                INSERT INTO public.bmstores(
                                date, retailer, "countryCode", currency, "sourceId", ean, brand, title, "shelfPrice",
                                "wasPrice", "cardPrice", "inStock", "onPromo", "promoData", "skuURL", "imageURL",
                                bundled, "masterSku", "priceValidUntil", "itemCondition", description, seller,
                                category)
                                VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s,
                                %s, %s, %s);
                                """

                        row_data = [date, RETAILER, country_code, currency, source_id, ean, brand, title, shelf_price,
                                    was_price, card_price, in_stock, on_promo, json.dumps(promo_data), full_url,
                                    image_url, bundled, master_sku, price_valid_until, item_condition, description,
                                    seller, category]

                        Utils.insert_to_db(insert_query, row_data, RETAILER)
        return True, item_counter
    except Exception as e:
        # Log to DBÃŸ
        log_error.error_db(RETAILER, e, sys.exc_info())
        # Log to file
        log_error.error_file(RETAILER, e, sys.exc_info())
        return False




[docs]def get_products_urls(url):
    """
    Navigate through all category URL's
    and add the product URL's in a list

    :param url: Category URLS
    :return: Product URLS
    :rtype: List
    """
    product_urls = []
    full_url = BASE_URL + url
    r = Utils.make_request(full_url, HEADERS, False, RETAILER)
    if r and r.status_code == 200:
        soup = BeautifulSoup(r.text, "lxml")
        if soup:
            try:
                pagination_div = soup.find("div", {"class": "pagination-links-top"})
                pagination = pagination_div.find("span", {"class": "text-nowrap"})
            except:
                pagination = None
            if pagination:
                pagination = int(pagination.text.replace("Page 1 of ", ""))
                for product_url in range(1, pagination + 1):
                    prod_full_url = f"{full_url}?page={product_url}&hitsPerPage=72"
                    r = Utils.make_request(prod_full_url, HEADERS, False, RETAILER)
                    if r and r.status_code == 200:
                        soup = BeautifulSoup(r.text, "lxml")
                        prod_listing = soup.find("div", "prod-listing")
                        if prod_listing:
                            product_list = prod_listing.find_all("a", {"class": "bm-product-stretch-link"})
                            if product_list:
                                for p_url in product_list:
                                    product_urls.append(p_url["href"])
            else:
                prod_listing = soup.find("div", "prod-listing")
                if prod_listing:
                    product_list = prod_listing.find_all("a", {"class": "bm-product-stretch-link"})
                    if product_list:
                        for p_url in product_list:
                            product_urls.append(p_url["href"])
    return product_urls




[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    navigation_url_count = len(urls)
    for url in urls:
        products_urls = get_products_urls(url["url"])
        result = get_data(products_urls)
        if result:
            url["success"] = result[0]
            item_counter_actual = result[1]
        else:
            item_counter_actual = 0

    item_counter += item_counter_actual

    duration = time.time() - start
    # # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter, duration, urls,
                          RETAILER, TABLE_NAME)
