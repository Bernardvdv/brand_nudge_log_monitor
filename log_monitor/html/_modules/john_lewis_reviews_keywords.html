[docs]def extract_keywords(text):
    """
    In this code, we first tokenize the input text into words using word_tokenize function from nltk.
    We then remove stop words and punctuation marks using stopwords module and a list comprehension.

    Next, we calculate the frequency distribution of the remaining words using FreqDist function from nltk.
    Finally, we extract the main keywords from the frequency distribution by selecting words with a length
    longer than 4 characters and return them as a set.

    Note that the output keywords are in lower case and in a set data structure, which removes duplicates.
    You can adjust the code to your needs, such as changing the minimum keyword length or returning a
    list instead of a set.
    :param text:
    :return:
    """
    # Tokenize text into words
    words = word_tokenize(text.lower())

    # Remove stop words and punctuation marks
    stop_words = set(stopwords.words('english'))
    words = [word for word in words if word.isalpha() and word not in stop_words]

    # Get frequency distribution of words
    freq_dist = nltk.FreqDist(words)

    # Extract main keywords with length longer than 4 characters
    keywords = list([word for word, freq in freq_dist.most_common() if len(word) > 4])

    return keywords

    """
    Makes a request to each URL and store the collected info in the database

    :param url: Product url

    :return: True if successful and product count
    :rtype: Boolean and integer

    try:
        item_counter = 0
        r = Utils.make_request(url, HEADERS, PROXIES_RES, RETAILER)
        if r and r.status_code == 200:
            soup = BeautifulSoup(r.text, "lxml")
            last_pagination_div = soup.find("ul", attrs={"data-test": "pagination"})
            if last_pagination_div:
                pagination_last = last_pagination_div.find_all("a", attrs={"data-test": "page-btn"})[-1].text
            else:
                pagination_last = 1
            for page in range(1, int(pagination_last) + 1):
                new_page_url = f"{url}?page={str(page)}"
                r = Utils.make_request(new_page_url, HEADERS, PROXIES_RES, RETAILER)
                if r and r.status_code == 200:
                    soup = BeautifulSoup(r.text, "lxml")
                    product_grid = soup.find("div", {"data-test-id": "product-grid"})

                    if product_grid:
                        # Get the schema
                        schema_products = product_grid.find_all("script")
                        # Convert schema to string for sanitization
                        schema_products_str = str(schema_products)

                        cleaned_schema_products = schema_products_str.replace(
                            '<script type="application/ld+json">', ""
                        ).replace("</script>", "")

                        # Convert data back to nested list
                        product_list = json.loads(cleaned_schema_products)
                        products = product_list[0]["itemListElement"]

                        for product in products:
                            product_url = product["url"]
                            r = Utils.make_request(product_url, HEADERS, PROXIES_RES, RETAILER, True)
                            if r and r.status_code == 200:
                                soup = BeautifulSoup(r.text, "lxml")

                                schema_products = soup.find_all("script", type='application/ld+json')
                                # Because I keep forgetting, this is to find the product schema
                                product_schema = Utils.get_product_schema(schema_products, "sku")
                                if product_schema:
                                    try:
                                        product_list = json.loads(product_schema)
                                    except:
                                        product_list = None

                                    source_id = product_list.get("sku")
                                    title = product_list.get("name", None)
                                    # description = Utils.remove_html_tags(product_list.get("description", None))
                                    keywords = extract_keywords(f"{title}")

                                    product_id = product_list.get("productId", None)
                                    review = get_reviews(product_id)
                                    if review["reviews"]:
                                        overall_rating = review.get("overall_rating", None)

                                        for rev in review["reviews"]:
                                            review_title = rev.get("title", None)
                                            review_text = rev.get("text", None)
                                            review_date_mil = rev.get("date", None)
                                            timestamp = review_date_mil / 1000  # convert milliseconds to seconds
                                            review_date = d.datetime.fromtimestamp(timestamp)

                                            review_source = rev.get("externalSource", None)
                                            username = rev['reviewer'].get("userName", None)
                                            location = rev['reviewer'].get("location", None)
                                            comments = rev.get("comments", None)
                                            rating_value = rev["overallRating"].get("value", None)
                                            rating_label = rev["overallRating"].get("label", None)
                                            rating_overall_value = rev["overallRating"].get("value", None)
                                            rating_questions = rev.get("ratingQuestions", None)
                                            review_id = rev.get("id", None)

                                            country_code = "UK"

[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0

    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data_from_search_query(url["url"], url["text"])
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter, duration, urls, RETAILER, TABLE_NAME)
