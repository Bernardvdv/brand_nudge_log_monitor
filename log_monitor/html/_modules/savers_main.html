[docs]def get_navigation_urls():
    """
    Get all navigation URLs from the base URL

    :return: All navigation URLs
    :rtype: Dictionary
    """
    item_list = []
    r = Utils.make_request(BASE_URL, custom_headers, PROXIES_RES, retailer)
    if r and r.status_code == 200:
        soup = BeautifulSoup(r.text, "lxml")
        nav_bar = soup.find('li', {"class": "shopcategories-class"}).find_all("a")

        for item in nav_bar:
            cat_url = item['href']
            if cat_url:
                url_list_item = {
                    "url": cat_url,
                    "category": item.text,
                    "attempts": 0,
                    "max_attempts": MAX_RETRIES,
                    "success": False
                }
                item_list.append(url_list_item)
    return item_list

    """
    Makes a request to each URL and store the collected info in the database

    :param url: Product url
    :return: True if successful and product count
    :rtype: Boolean and integer
    """
    try:
        r = Utils.make_request(BASE_URL + url, custom_headers, PROXIES_RES, retailer)
        if r and r.status_code == 200:
            soup_cat = BeautifulSoup(r.text, "lxml")
            last_pagination_value = soup_cat.find("li", {"class": "lastpage"}).find("a").text
            item_counter = 0
            for page in range(0, int(last_pagination_value)):
                full_url = f"{BASE_URL}{url}?page={page}&resultsForPage=100"
                r = Utils.make_request(full_url, custom_headers, PROXIES_RES, retailer)
                if r and r.status_code == 200:
                    product_page = BeautifulSoup(r.text, "lxml")

                    if product_page.find("div", {"class": "plp__product-list-container"}) is not None:
                        product_container = product_page.find("div", {"class": "plp__product-list-container"}).find_all("a")
                        product_container_without_duplicates = list(set(product_container))

                        for product_url in product_container_without_duplicates:
                            if product_url.get("href", None) is not None:
                                full_product_url = BASE_URL + product_url.get("href", None)
                                r = Utils.make_request(full_product_url, custom_headers, PROXIES_RES, retailer)
                                if r and r.status_code == 200:
                                    product_item_page = BeautifulSoup(r.text, "lxml")
                                    schema_products = product_item_page.find_all("script", type='application/ld+json')
                                    # Because I keep forgetting, this is to find the product schema
                                    product_schema = Utils.get_product_schema(schema_products, "sku")
                                    if product_schema:
                                        try:
                                            product_list = json.loads(product_schema)
                                        except:
                                            product_list = None

                                        if product_list and type(product_list) is dict:
                                            try:
                                                title = product_list.get("name", None)
                                            except:
                                                title = None
                                            try:
                                                shelfPrice = product_list["offers"].get("price", None)
                                            except:
                                                shelfPrice = None
                                            try:
                                                currency = product_list["offers"].get("priceCurrency", None)
                                            except:
                                                currency = None
                                            try:
                                                sourceId = product_list.get("sku", None)
                                            except:
                                                sourceId = None
                                            try:
                                                imageURL = product_list.get("image", None)[0]
                                            except:
                                                imageURL = None
                                            try:
                                                brand = product_list["brand"].get("name", None)
                                            except:
                                                brand = None
                                            try:
                                                ean = product_list["offers"].get("gtin8", None)
                                            except:
                                                ean = None
                                            try:
                                                description = product_list.get("description", None)
                                            except:
                                                description = None
                                            try:
                                                inStock_schema = product_list["offers"].get("availability", None)
                                                if "InStock" in inStock_schema:
                                                    inStock = True
                                                else:
                                                    inStock = False
                                            except:
                                                inStock = None
                                            try:
                                                skuURL = product_list["offers"].get("url")
                                            except:
                                                skuURL = None

                                            category_div = product_item_page.find("div", {"class": "breadcrumb"})

                                                if "Home" not in href.text:

[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    navigation_url_count = 0

    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(url["url"])
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter, duration, urls, retailer, TABLE_NAME)
