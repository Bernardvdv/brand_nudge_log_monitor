    """
    Get the promotions form the schema
    :param schema: Product schema
    :return: List of promotions
    """
    cleaned_schema_products = schema.replace(
        '<script id="__NEXT_DATA__" type="application/json">', ""
    ).replace("</script>", "")

    product_list = json.loads(cleaned_schema_products)
    promotion = product_list["props"]["pageProps"]["product"]["variants"][0]["messaging"]
    promotions = []
    for pr in promotion:
        if pr["type"] == "promotional":
            promo_text = pr["title"]
            promotions.append(promo_text)
    return promotions

[docs]def get_review_detail(schema):
    """
    Get the product review detail form the schema
    :param schema: Product schema
    :return: review count if found else None
    """
    cleaned_schema_products = schema.replace(
        '<script id="__NEXT_DATA__" type="application/json">', ""
    ).replace("</script>", "")

    product_list = json.loads(cleaned_schema_products)

    try:
        if "product" in product_list[0]["props"]["pageProps"] and "reviewSummary" in product_list[0]["props"]["pageProps"]["product"]:
            try:
                review_count = product_list[0]["props"]["pageProps"]["product"]["reviewSummary"].get("numberOfReviews")
            except:
                review_count = None
            try:
                product_rating = product_list[0]["props"]["pageProps"]["product"]["reviewSummary"].get("averageRating")
            except:
                product_rating: None
        elif "skus" in product_list[0]["props"]["pageProps"] and "reviewData" in product_list[0]["props"]["pageProps"]:
            try:
                review_count = product_list[0]["props"]["pageProps"]["reviewData"].get("numberOfReviews")
            except:
                review_count = None
            try:
                product_rating = product_list[0]["props"]["pageProps"]["reviewData"].get("averageRating")
            except:
                product_rating = None
        else:
            review_count = None
            product_rating = None
    except Exception as e:
        log_error.error_file(retailer, e, sys.exc_info())

    return review_count, product_rating

    """ Get all navigation URLs from the state API
    :returns URLs as a dictionary
    """
    r = Utils.make_request(JOHN_LEWIS_STATE_URL, headers, PROXIES_RES, retailer)
    soup = BeautifulSoup(r.text, "html.parser")
    categories_content = json.loads(soup.contents[0])
    categories_children = categories_content["categories"]['child']

    item_list = list()

    for child in categories_children:
        if child['name'] == "Electricals" and child["type"] == "DEPARTMENT":
            for url in child["child"]:
                for u in url["child"]:
                    url_list_item = {
                        "url": u["url"],
                        "text": f'Electricals/{u["name"]}',
                        "attempts": 0,
                        "max_attempts": MAX_RETRIES,
                        "success": False
                     }
                    item_list.append(url_list_item)

    for child in categories_children:
        if child['name'] == "Home & Garden" and child["type"] == "DEPARTMENT":
            for url in child["child"]:
                for u in url["child"]:
                    url_list_item = {
                        "url": u["url"],
                        "text": f'Home & Garden/{u["name"]}',
                        "attempts": 0,
                        "max_attempts": MAX_RETRIES,
                        "success": False
                    }
                    item_list.append(url_list_item)
    return item_list

[docs]def parse_urls(urls):
    """
    Create BS4 object from url
    :param urls:
    :return:
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0

    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        full_url = JOHN_LEWIS_BASE_URL + url["url"]
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(full_url, url["text"])
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter, duration, urls, retailer, TABLE_NAME)
