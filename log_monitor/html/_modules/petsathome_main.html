[docs]def get_brand(soup):
    """
    Get the product brand from the product page

    :param soup: BS4 object of product page
    :return: brand if found, else none
    :rtype: String
    """
    try:
        schema_products = soup.find_all("script")
        for schema in schema_products:
            if "variant" in str(schema):
                pattern = re.compile(r'"brand"\s*:\s*"(.+?)"')
                match = pattern.search(str(schema))
                if match:
                    brand = match.group(1)
                    return brand
                else:
                    return None
    except Exception as e:
        # Log to DB
        log_error.error_db(RETAILER, e, sys.exc_info())
        # Log to file
        log_error.error_file(RETAILER, e, sys.exc_info())
        return None

    """
    Get max pagination number from products page

    :param url: Products url
    :return: Max pagination number
    :rtype: Integer
    """
    full_url = f"{url}?currentPage=1&pageSize=72&orderBy=1"
    r = Utils.make_request(full_url, HEADERS, PROXIES_RES, RETAILER, True)
    if r and r.status_code == 200:
        soup = BeautifulSoup(r.text, "lxml")
        if soup:
            try:
                return int(soup.find("a", {"class": "pagination__link--last"})["data-page"])
            except:
                return 1
        else:
            return 1
    else:
        return 1




[docs]def get_data(category_url, main_category):
    """
    Makes a request to each URL and store the collected info in the database

    :param sidenav_urls: Product url
    :param main_category: Product category
    :return: True if successful and product count
    :rtype: Boolean and integer
    """
    try:
        item_counter = 0

        country_code = "UK"
        currency = "GBP"
        date = datetime.now()
        card_price = None
        bundled = None
        master_sku = None

        insert_query = """
            INSERT INTO public.petsathome(
                date, retailer, "countryCode", currency, "sourceId", ean, brand, title, "shelfPrice",
                "wasPrice", "cardPrice", "inStock", "onPromo", "promoData", "skuURL", "imageURL",
                bundled, "masterSku", category, "partNumber",  "overLayText", "hasVariants", size, weight)
                VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s,
                %s, %s, %s, %s, %s);
            """

        last_pagination_number = get_pagination(category_url)
        for page in range(1, last_pagination_number + 1):
            full_url = f"{category_url}?currentPage={page}&pageSize=72&orderBy=1"
            r = Utils.make_request(full_url, HEADERS, PROXIES_RES, RETAILER, True)
            if r and r.status_code == 200:
                soup = BeautifulSoup(r.text, "lxml")
                products_grid = soup.find("section", {"class": "layout-product-tile-holder"})
                sub_category = soup.find("h2", {"class": "content-container__heading-1"})
                if sub_category:
                    sub_category_text = sub_category.text
                if sub_category_text not in main_category:
                    category = f"{main_category}/{sub_category_text}"
                else:
                    category = main_category

                if products_grid:
                    product_info_div = products_grid.find_all("div",
                                                              {"class": "-js-ProductInfoPostAjaxJavascript"})

                    for info in product_info_div:
                        script_tag = info.find("script", type="text/javascript")
                        try:
                            if script_tag:
                                script_tag = str(script_tag)
                                if "shortDescription" in script_tag:
                                    second_brace_pos = script_tag.find("{", script_tag.find("{") + 1)
                                    cleaned_string = script_tag[second_brace_pos:].replace(";", '') \
                                        .replace(";", '').replace("</script>", "").replace("})", "")
                                    cleaned_product = json.loads(cleaned_string)
                                    if cleaned_product:
                                        if BASE_URL not in cleaned_product.get("productDetailsPageURL"):
                                            sku_url = BASE_URL + cleaned_product.get("productDetailsPageURL")
                                        else:
                                            sku_url = cleaned_product.get("productDetailsPageURL")
                                        overlay_text = cleaned_product.get("overlayText", None)
                                        if "https://" not in sku_url:
                                            sku_url = f"https://{sku_url}"
                                        r = Utils.make_request(sku_url, HEADERS, PROXIES_RES, RETAILER, True)
                                        if r and r.status_code == 200:
                                            soup = BeautifulSoup(r.text, "lxml")
                                            if soup:
                                                brand = get_brand(soup)
                                                in_stock = get_stock(soup)
                                                was_price = get_wasprice(soup)
                                            else:
                                                brand = None
                                                in_stock = None
                                                was_price = None
                                        else:
                                            brand = None
                                            in_stock = None
                                            was_price = None

                                        has_variants = cleaned_product.get("hasVariants", None)
                                        if has_variants:
                                            attributes = cleaned_product["attrs"].get("weights", None)
                                            sizes = cleaned_product["attrs"].get("sizes", None)
                                            variants = cleaned_product.get("SKUDetails", None)
                                            for key, value in variants.items():
                                                title = value.get("name", None)
                                                source_id = key
                                                shelf_price = value["price"].get("displayPrice")[1:]
                                                on_promo = value["price"].get("isDiscountedPrice")
                                                promo_data = value["price"].get("promoTextList")

                                                if not promo_data:
                                                    promo_data = []

                                                image_url = f"//media.petsathome.com{value['fullImage']}"
                                                part_number = value.get("partnumber", None)
                                                if sizes:
                                                    for s in sizes:
                                                        if s["partnumber"] == part_number:
                                                            size = s["value"]
                                                else:
                                                    size = None
                                                if attributes:
                                                    for attr in attributes:
                                                        if attr["partnumber"] == part_number:
                                                            weight = attr["value"]
                                                else:
                                                    weight = None

                                                item_counter += 1

                                                pattern = re.compile(r"\d+\s*x")
                                                matches = pattern.findall(title)

                                                if len(matches) > 0:
                                                    ean = "SET"
                                                else:
                                                    ean = None

                                                row_data = [date, RETAILER, country_code, currency, source_id,
                                                            ean, brand, title, shelf_price, was_price, card_price,
                                                            in_stock, on_promo, json.dumps(promo_data), sku_url,
                                                            image_url, bundled, master_sku, category, part_number,
                                                            overlay_text, has_variants, size, weight]

                                                Utils.insert_to_db(insert_query, row_data, RETAILER)
                                        else:
                                            title = cleaned_product.get("title", None)
                                            source_id = cleaned_product.get("singleSKUCatentryId", None)
                                            shelf_price = cleaned_product["price"].get("displayPrice")[1:]
                                            on_promo = cleaned_product["price"].get("isDiscountedPrice")
                                            promo_data = cleaned_product["price"].get("promoTextList")

                                            pattern = re.compile(r"\d+\s*x")
                                            matches = pattern.findall(title)

                                            if len(matches) > 0:
                                                ean = "SET_"
                                            else:
                                                ean = None

                                            if not promo_data:
                                                promo_data = []

                                            image_url = f"//media.petsathome.com{cleaned_product['fullImage']}"
                                            part_number = cleaned_product.get("partnumber", None)
                                            size = None
                                            weight = None

                                            item_counter += 1

                                            row_data = [date, RETAILER, country_code, currency, source_id, ean,
                                                        brand, title, shelf_price, was_price, card_price, in_stock,
                                                        on_promo, json.dumps(promo_data), sku_url, image_url,
                                                        bundled, master_sku, category, part_number, overlay_text,
                                                        has_variants, size, weight]
                                            Utils.insert_to_db(insert_query, row_data, RETAILER)
                        except:
                            pass
        return True, item_counter
    except Exception as e:
        # Log to DB
        log_error.error_db(RETAILER, e, sys.exc_info())
        # Log to file
        log_error.error_file(RETAILER, e, sys.exc_info())
        return False
