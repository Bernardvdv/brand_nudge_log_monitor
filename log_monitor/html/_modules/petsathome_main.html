    """
    Get all urls for the specific category from the sidenav

    :param url: Category url
    :return: Product URL's
    :rtype: List
    """
    navigation_urls = []
    r = Utils.make_request(url, headers, PROXIES_RES, retailer, True)
    if r and r.status_code == 200:
        soup = BeautifulSoup(r.text, "lxml")
        sidenav = soup.find("nav", {"class": "side-nav-category"})

            nav_urls_a = sidenav.find_all("a")

    return navigation_urls

[docs]def get_brand(soup):
    """
    Get the product brand from the product page

    :param soup: BS4 object of product page
    :return: brand if found, else none
    :rtype: String
    """
    try:
        schema_products = soup.find_all("script")
        for schema in schema_products:
            if "variant" in str(schema):
                pattern = re.compile(r'"brand"\s*:\s*"(.+?)"')
                match = pattern.search(str(schema))
                if match:
                    brand = match.group(1)
                    return brand
                else:
                    return None
    except Exception as e:
        # Log to DB
        log_error.error_db(retailer, e, sys.exc_info())
        # Log to file
        log_error.error_file(retailer, e, sys.exc_info())
        return None

    """
    Get max pagination number from products page

    :param url: Products url
    :return: Max pagination number
    :rtype: Integer
    """
    full_url = f"{url}?currentPage=1&pageSize=72&orderBy=1"
    r = Utils.make_request(full_url, headers, PROXIES_RES, retailer, True)
    if r and r.status_code == 200:
        soup = BeautifulSoup(r.text, "lxml")
        if soup:
            try:
                return int(soup.find("a", {"class": "pagination__link--last"})["data-page"])
            except:
                return 1
        else:
            return 1
    else:
        return 1

    """
    Makes a request to each URL and store the collected info in the database

    :param sidenav_urls: Product url
    :param category: Product category
    :return: True if successful and product count
    :rtype: Boolean and integer
    """
    try:
        item_counter = 0

        countryCode = "UK"
        currency = "GBP"
        date = datetime.now()
        cardPrice = None
        bundled = None
        masterSku = None
        ean = None

        insert_query = """
                            INSERT INTO public.petsathome(
                                date, retailer, countrycode, currency, "sourceId", ean, 
                                brand, title, "shelfPrice", "wasPrice", "cardPrice", 
                                "inStock", "onPromo", "promoData", "skuURL", "imageURL", 
                                bundled, "masterSku", category, "partNumber", 
                                "overLayText", "hasVariants", size, weight)
                                VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, 
                                %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s);
                        """

        for url in sidenav_urls:
            last_pagination_number = get_pagination(url)
            for page in range(1, last_pagination_number + 1):
                full_url = f"{url}?currentPage={page}&pageSize=72&orderBy=1"
                r = Utils.make_request(full_url, headers, PROXIES_RES, retailer, True)
                if r and r.status_code == 200:
                    soup = BeautifulSoup(r.text, "lxml")
                    products_grid = soup.find("section", {"class": "layout-product-tile-holder"})
                    sub_category = soup.find("h2", {"class": "content-container__heading-1"})
                    if sub_category:
                        sub_category = sub_category.text
                    if sub_category and sub_category not in category:
                        category = f"{category}/{sub_category}"

                    if products_grid:
                        product_info_div = products_grid.find_all("div", {"class": "-js-ProductInfoPostAjaxJavascript"})

                        for info in product_info_div:
                            script_tag = info.find("script", type="text/javascript")
                            try:
                                if script_tag:
                                    script_tag = str(script_tag)
                                    if "shortDescription" in script_tag:
                                        second_brace_pos = script_tag.find("{", script_tag.find("{") + 1)
                                        cleaned_string = script_tag[second_brace_pos:].replace(";", '')\
                                            .replace(";", '').replace("</script>", "").replace("})", "")
                                        cleaned_product = json.loads(cleaned_string)
                                        if cleaned_product:
                                            skuURL = BASE_URL + cleaned_product.get("productDetailsPageURL")
                                            overLayText = cleaned_product.get("overlayText", None)

                                            r = Utils.make_request(f"https://{skuURL}", headers, PROXIES_RES,
                                                                   retailer, True)
                                            if r and r.status_code == 200:
                                                soup = BeautifulSoup(r.text, "lxml")
                                                if soup:
                                                    brand = get_brand(soup)
                                                    inStock = get_stock(soup)
                                                    wasPrice = get_wasprice(soup)
                                                else:
                                                    brand = None
                                                    inStock = None
                                                    wasPrice = None
                                            else:
                                                brand = None
                                                inStock = None
                                                wasPrice = None

                                            hasVariants = cleaned_product.get("hasVariants", None)
                                            if hasVariants:
                                                attributes = cleaned_product["attrs"].get("weights", None)
                                                sizes = cleaned_product["attrs"].get("sizes", None)
                                                variants = cleaned_product.get("SKUDetails", None)
                                                for key, value in variants.items():
                                                    title = value.get("name", None)

                                                    shelfPrice = value["price"].get("displayPrice")[1:]
                                                    onPromo = value["price"].get("isDiscountedPrice")

                                                    partNumber = value.get("partnumber", None)

                                                    else:

                                                    item_counter += 1

                                                    row_data = [date, retailer, countryCode, currency, sourceId,

                                            else:
                                                title = cleaned_product.get("title", None)

                                                shelfPrice = cleaned_product["price"].get("displayPrice")[1:]
                                                onPromo = cleaned_product["price"].get("isDiscountedPrice")

[docs]def get_navigation_urls():
    """
    Create a nested dictionary of category URL's

    :return: All navigation URLs
    :rtype: Dictionary
    """
    item_list = []
    urls = [
        {"category": "Dog", "url": "https://www.petsathome.com/shop/en/pets/dog"},
        {"category": "Cat", "url": "https://www.petsathome.com/shop/en/pets/cat"},
        {"category": "Small Animal", "url": "https://www.petsathome.com/shop/en/pets/small-pet"},
        {"category": "Fish", "url": "https://www.petsathome.com/shop/en/pets/fish"},
        {"category": "Reptile", "url": "https://www.petsathome.com/shop/en/pets/reptile"},
        {"category": "Bird & Wildlife", "url": "https://www.petsathome.com/shop/en/pets/bird-and-wildlife"},
    ]

    for item in urls:
        url_list_item = {
            "url": item["url"],
            "category": item["category"],
            "attempts": 0,
            "max_attempts": MAX_RETRIES,
            "success": False
        }
        item_list.append(url_list_item)
    return item_list




[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            sidenav_urls = get_sidenav_urls_per_category(url["url"])
            result = get_data(sidenav_urls, url["category"])

            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter,
                          navigation_url_count,
                          item_counter,
                          duration,
                          urls,
                          retailer,
                          TABLE_NAME)
