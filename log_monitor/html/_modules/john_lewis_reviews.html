    """
    Get reviews for the current product.
    :param product_id: Product ID
    :return: Reviews as list object
    """
    try:
        url = f"https://api.johnlewis.com/ratings-reviews/v1/reviews" \
              f"?productId={product_id}" \
              f"&pageNumber=1" \
              f"&pageSize=10" \
              f"&sort=newest" \
              f"&api_key=AIzaSyDrfSLzq-WxgyLPSKu53oW7y4TJo7bQdpY"
        # Proxy sometimes times out, so we attempt 5 times before exit
        success = False
        try:
            for x in range(5):
                r = Utils.make_request(url, api_headers, False, retailer)
                if r is not None:
                    if r.status_code == 200:
                        success = True
                        break
        except Exception as e:
            # Log to DB
            log_error.error_db(retailer, e, sys.exc_info())
            # Log to file
            log_error.error_file(retailer, e, sys.exc_info())
            pass

        if success:
            # print(f"Reviews Status Code: {r.status_code}")
            data = r.json()
            if data["summary"]["totalResults"] > 0:
                overall_rating = data["overallRatings"].get("label", None)
                reviews = data.get("reviews", None)
                return {"overall_rating": overall_rating, "reviews": reviews}
        return {"overall_rating": None, "reviews": None}
    except Exception as e:
        return {"overall_rating": None, "reviews": None}

[docs]def get_navigation_urls():
    """ Get all navigation URLs from the state API
    :returns URLs as a dictionary
    """
    r = Utils.make_request(JOHN_LEWIS_STATE_URL, headers, PROXIES_RES, retailer)
    soup = BeautifulSoup(r.text, "html.parser")
    categories_content = json.loads(soup.contents[0])
    categories_children = categories_content["categories"]['child']

    item_list = list()

    for child in categories_children:
        if child['name'] == "Electricals" and child["type"] == "DEPARTMENT":
            for url in child["child"]:
                for u in url["child"]:
                    url_list_item = {
                        "url": u["url"],
                        "text": "Electricals",
                        "attempts": 0,
                        "max_attempts": MAX_RETRIES,
                        "success": False
                     }
                    item_list.append(url_list_item)

    for child in categories_children:
        if child['name'] == "Home & Garden" and child["type"] == "DEPARTMENT":
            for url in child["child"]:
                for u in url["child"]:
                    url_list_item = {
                        "url": u["url"],
                        "text": "Home & Garden",
                        "attempts": 0,
                        "max_attempts": MAX_RETRIES,
                        "success": False
                    }
                    item_list.append(url_list_item)
    return item_list




[docs]def parse_urls(urls):
    """
    Create BS4 object from url
    :param urls:
    :return:
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0

    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        time.sleep(1)
        full_url = JOHN_LEWIS_BASE_URL + url["url"]
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            result = get_data(full_url, url["text"])
            if result:
                url["success"] = result[0]
                item_counter_actual = result[1]
            else:
                item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter, navigation_url_count, item_counter, duration, urls, retailer, TABLE_NAME)
