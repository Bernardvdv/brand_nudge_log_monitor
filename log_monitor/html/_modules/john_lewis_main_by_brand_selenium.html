"""John Lewis product main scraper file."""
import json
import time

import requests
from bs4 import BeautifulSoup
from fake_useragent import UserAgent
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.by import By
from webdriver_manager.chrome import ChromeDriverManager

# Generate fake Request headers
ua = UserAgent()
headers = {"user-agent": ua.chrome}

# Initiate Selenium class
driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()))

JOHN_LEWIS_BRANDS_URL = "https://www.johnlewis.com/brands"



[docs]def get_navigation_urls():
    """ Get all navigation URLs from the state API
    :returns URLs as a dictionary
    """
    brand_list = list()

    driver.get(JOHN_LEWIS_BRANDS_URL)

    time.sleep(2)

    brand_div = driver.find_element(By.CLASS_NAME, "right").find_elements(By.TAG_NAME, 'a')
    for brand in brand_div:
        href = brand.get_attribute('href')
        if href is not None:
            brand_list.append(href)

    return brand_list




[docs]def parse_urls(urls):
    """
    Create BS4 object from url
    :param urls: List of brand urls
    :return: Soup object of the product grid
    """
    for url in urls:
        r = requests.get(
            url,
            headers=headers,
            timeout=20,
            allow_redirects=True
        )
        soup = BeautifulSoup(r.text, "html.parser")

        product_grid = soup.find("div", {"data-test-id": "product-grid"})

        # Get the schema
        schema_products = product_grid.find_all("script")
        # Convert schema to string for sanitization
        schema_products_str = str(schema_products)

        cleaned_schema_products = schema_products_str.replace(
            '<script type="application/ld+json">', ""
        ).replace("</script>", "")

        # Convert data back to nested list
        product_list = json.loads(cleaned_schema_products)

        products = product_list[0]["itemListElement"]

        url_list = list()
        for x in products:
            url_list.append(x["url"])

        for url in url_list:
            r = requests.get(
                url,
                headers=headers,
                timeout=20,
                allow_redirects=True
            )
            soup = BeautifulSoup(r.text, "html.parser")
            schema_products = soup.find_all("script")[0]

            # Convert schema to string for sanitization
            schema_products_str = str(schema_products)

            cleaned_schema_products = schema_products_str.replace(
                '<script type="application/ld+json">', ""
            ).replace("</script>", "")

            # Convert data back to nested list
            product_list = json.loads(cleaned_schema_products)

            try:
                p_name = product_list["name"]
            except:
                p_name = None
            try:
                p_prize = product_list["offers"]["price"]
            except:
                p_prize = None
            try:
                p_review_count = product_list["aggregateRating"]["reviewCount"]
            except:
                p_review_count = None
            try:
                p_rating = product_list["aggregateRating"]["ratingValue"]
            except:
                p_rating = None
            try:
                p_availability = product_list["offers"]["availability"]
            except:
                p_availability = None
            try:
                p_price_currency = product_list["offers"]["priceCurrency"]
            except:
                p_price_currency = None
            try:
                p_id = product_list["productId"]
            except:
                p_id = None
            try:
                p_sku = product_list["sku"]
            except:
                p_sku = None
            try:
                p_color = product_list["color"]
            except:
                p_color = None
            try:
                p_description = product_list["description"].replace("<p>", "").replace("</p>", "")
            except:
                p_description = None
            try:
                p_image_url = product_list["image"]
            except:
                p_image_url = None
            try:
                p_brand = product_list["brand"]["name"]
            except:
                p_brand = None

            data = list()
            data.append([p_name, p_prize, p_review_count, p_rating, p_availability, p_price_currency, p_id, p_sku,
                         p_color, p_description, p_image_url, p_brand])

    return soup
