[docs]def get_cleaned_product_urls(url):
    """
    Finds all the product hyperlinks and remove duplicates

    :param url: Category URL
    :return: All product urls
    :rtype: List
    """
    r = Utils.make_request(url, HEADERS, PROXIES, RETAILER, False)
    soup = BeautifulSoup(r.text, "lxml")
    product_grid = soup.find("div", {"class": "products"})
    if product_grid:
        items_grid = product_grid.find_all("div", {"class": "item"})
        if items_grid:
            grid_urls = []
            for item in items_grid:
                url = BASE_URL + item.find('a')['href']
                grid_urls.append(url)
            # Removing any duplicates
            return list(dict.fromkeys(grid_urls))
        return None
    return None




[docs]def get_variant_urls(url):
    """
    Request to item page and collect variant urls and remove any duplicates

    :param url: Product url
    :return: All variant urls
    :rtype: List
    """
    c = []
    r = Utils.make_request(url, HEADERS, PROXIES, RETAILER, False)
    if r and r.status_code == 200:
        soup = BeautifulSoup(r.text, "lxml")
        if soup:
            prod_grid = soup.find("div", {"class": "prod-highlight"})

            # if below is greater than 1, it probably means it's a product with different variants
            multi_product_check = prod_grid.find_all("div", {"class": "prod"})
            for item in multi_product_check:
                try:
                    urlhref = item.find("a", {"class": "click"})
                except:
                    urlhref = None

                if urlhref:
                    c.append(urlhref["href"])
                else:
                    c.append(url)
    return list(dict.fromkeys(c))

[docs]def parse_urls(urls):
    """
    Loop through all URL's and update the main URL dict when successfully parsed.
    Collects statistical data and store it and the end of the program

    :param urls: URL dictionary
    :return: None
    :rtype: None
    """
    start = time.time()
    navigation_counter = 0
    item_counter = 0
    for url in urls:
        navigation_counter += 1
        navigation_url_count = len(urls)
        while url["success"] is False and url["attempts"] < url["max_attempts"]:
            # Increment attempts counter
            url["attempts"] += 1
            # Cleaned_urls returns all the products urls on the page
            cleaned_urls = get_cleaned_product_urls(url["url"])
            if cleaned_urls:
                for clean_url in cleaned_urls:
                    # item_urls returns all the variants of each product url
                    item_urls = get_variant_urls(clean_url)
                    if item_urls:
                        for item in item_urls:
                            result = get_data(item)
                            if result:
                                url["success"] = result[0]
                                item_counter_actual = result[1]
                            else:
                                item_counter_actual = 0
                    else:
                        item_counter_actual = 0

        item_counter += item_counter_actual

    duration = time.time() - start
    # Lets record some stats
    Statistics.statistics(navigation_counter,
                          navigation_url_count,
                          item_counter,
                          duration,
                          urls,
                          RETAILER,
                          TABLE_NAME)
